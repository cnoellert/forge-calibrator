## Recipe

```python
"""Drop-in replacement for forge_flame.wiretap.extract_frame_bytes that uses
flame.PyExporter().export() instead of the wiretap_rw_frame CLI.

Snapshot output is raw working colour space (BakeViewTransform=false in
/opt/Autodesk/cfg/export_snapshot.cfg). PNG-8bit container — handled by
forge_core.image.buffer.decode_image_container() via its PNG magic-byte
branch, which returns uint8 RGB. No bit-depth-keyed channel-order swap is
needed; that's a Wiretap raw-buffer quirk. PNG is RGB-ordered.

Verified 2026-04-30 (260430-e5y): 0.41s round-trip on 20x20 test plate;
output file decodes to (20,20,3) uint8 via decode_image_container.
"""
from __future__ import annotations
import os, shutil, tempfile
from typing import Optional, Tuple

# Pinned to Flame 2026.2.1; consider glob-resolving across versions:
#   /opt/Autodesk/presets/*/export/presets/flame/file_sequence/PNG/PNG (8-bit).xml
_SNAPSHOT_PRESET = "/opt/Autodesk/presets/2026.2.1/export/presets/flame/file_sequence/PNG/PNG (8-bit).xml"

def extract_frame_via_export(
    clip, target_frame: Optional[int] = None,
) -> Optional[Tuple[bytes, int, int, int]]:
    """Export one frame via PyExporter. Returns (bytes, w, h, bit_depth) or None.
    Accepts PyClip (direct w/h/bit_depth attrs) or PyClipNode (resolution PyAttribute).
    target_frame is advisory — TODO(next-spike): park player before export."""
    import flame
    if not os.path.isfile(_SNAPSHOT_PRESET): return None
    try:                                            # PyClip path
        w, h, bd = int(clip.width), int(clip.height), int(clip.bit_depth)
    except (AttributeError, TypeError):
        try:                                        # PyClipNode path
            r = clip.resolution.get_value()
            w, h, bd = int(r.width), int(r.height), int(r.bit_depth)
        except Exception: return None
    out_dir = tempfile.mkdtemp(prefix="forge_export_")
    try:
        e = flame.PyExporter(); e.foreground = True
        e.export(clip, _SNAPSHOT_PRESET, out_dir)
        files = sorted(f for f in os.listdir(out_dir) if not f.startswith("."))
        if not files: return None
        with open(os.path.join(out_dir, files[0]), "rb") as f:
            return f.read(), w, h, bd
    except Exception: return None
    finally: shutil.rmtree(out_dir, ignore_errors=True)
```

LOC count: 25 lines for the function (def → finally), well under the 30-LOC budget. Signature shape-matches `forge_flame.wiretap.extract_frame_bytes(clip, target_frame=None) -> Optional[Tuple[bytes, int, int, int]]`. Colour-space note in module docstring. Dual-shape clip handling (PyClip and PyClipNode). One `TODO(next-spike)` for player-frame-parking — see Next Planning Step.
